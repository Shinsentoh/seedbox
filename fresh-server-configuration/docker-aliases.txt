alias all-logs="docker ps -q | xargs -P 13 -L 1 docker logs --follow"
alias dockerStopAll="docker stop $(docker ps -a -q)"
alias dockerStartAll="docker start $(docker ps -a -q)"
alias dockertruncate="sh -c 'truncate -s 0 /home/docker/containers/*/*-json.log'"

function dockerup () { docker-compose $(find docker-compose.* | sed -e 's/^/-f /') up --remove-orphans }

function dockerdown () {
    docker-compose $(find docker-compose.* | sed -e 's/^/-f /') down --remove-orphans
}
